#include <stdio.h>
int square (int);
int prime (int);
int even  (int);

int (*func_ptr) (int);

int main (void)
	{
	int choice, num, ret;

	printf ("Enter the choice to perform the below operations:\n 1. Square of a Value\n	\
	2. Check Odd/Even\n3. Check Prime number\n");
	scanf ("%d", &choice);
	printf ("Enter the number\n");
	scanf ("%d", &num);

	switch (choice)
		{
		case 1:
			func_ptr = square;
			printf ("Performing Square...");
		break;

		case 2:
			func_ptr = even;
			printf ("Performing Odd/Even Check...");
		break;

		case 3:
			func_ptr = prime;
			printf ("Performing Prime Check...");
		break;

		default:
			printf ("Sorry, wrong input!!\n");
		break;
		}

	ret = func_ptr (num);
	printf ("Result: %d", ret);

	return 0;
	}
        int square (int a )
       {
        printf ("the area of square is %d", a*a);
       }
       

       int even (int a)
       {
         if (a % 2 == 0)
         {
           printf ("the given number is even");
         }
          else
         {
           printf ("the given number is odd");
         }
       }
       int prime (int a)
       {
        int i , count;
        count = 0;
        for (i = 1; i <= a; i++)
        {
          if (a % i == 0)
          {
          ++count;
          }
        }

          if (count == 2)
         {
         printf ("given number is prime");
         }
         else
         {
         printf ("given number is not a prime");
         }
       }
